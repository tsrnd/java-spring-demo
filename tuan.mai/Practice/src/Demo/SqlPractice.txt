CREATE PROCEDURE gradeDistribution()
BEGIN
	select Name, ID from Grades
    where Final > Midterm1 * 0.25 + Midterm2 * 0.25 + Final * 0.5
        and Final > Midterm1 * 0.5 + Midterm2 * 0.5
    order by left(Name,3);
END
